git init - инициализация гита, создание репозитория
git push - отправка коммитов в удаленный репозитори
git commit - сохранение изменений
git pull - получение и слияние измениений
git branch - управление веткой
git status - получение текуцего состояния индекса и каталога
git clone - клонирование удалленого репозитория
git cheackout - переключение на определенные коммиты
git switch - переключение между ветками
git log - логирование истории коммитов
